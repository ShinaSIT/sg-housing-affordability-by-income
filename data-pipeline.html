CPI by Household Income Group (Annual, Group-Specific)

load_cpi_annual <- function(filepath, group_label) {
  raw <- read_excel(filepath, sheet = 1, skip = 10, n_max = 2)
  colnames(raw)[1] <- "Series"
  
  raw %>%
    filter(str_trim(Series) == "All Items") %>%
    pivot_longer(cols = -Series, names_to = "Year", values_to = "CPI") %>%
    mutate(
      Year = as.integer(str_trim(Year)),
      CPI  = as.numeric(CPI),
      Group = group_label
    ) %>%
    filter(!is.na(Year), !is.na(CPI), Year >= 2000, Year <= 2025) %>%
    select(Group, Year, CPI)
}

Data Cleaning and Standardisation

# Aggregate deciles into 3 income groups
income_grouped <- income_long %>%
  mutate(
    Group = case_when(
      Decile %in% c("1st (Lowest)", "2nd")                     ~ "Bottom 20%",
      Decile %in% c("3rd", "4th", "5th", "6th", "7th", "8th") ~ "Middle 60%",
      Decile %in% c("9th", "10th (Highest)")                   ~ "Top 20%"
    )
  ) %>%
  filter(!is.na(Group)) %>%
  group_by(Group, Year) %>%
  summarise(Income_Nominal = mean(Income_Nominal, na.rm = TRUE), .groups = "drop")

# Compute annual median HDB resale price
hdb_annual <- hdb_raw %>%
  mutate(
    Year = as.integer(str_extract(month, "\\d{4}")),
    resale_price = as.numeric(resale_price)
  ) %>%
  filter(Year >= 2000, Year <= 2025) %>%
  group_by(Year) %>%
  summarise(HDB_Median_Price = median(resale_price, na.rm = TRUE), .groups = "drop")

Final Tidy Dataset

# Income in long format
income_long_final <- income_indexed %>%
  select(Year, Income_Group = Group, Index_Value = Income_Index) %>%
  mutate(Measure_Type = "Real Income")

# HDB replicated for each income group
hdb_long_final <- hdb_indexed %>%
  select(Year, Index_Value = HDB_Index) %>%
  crossing(Income_Group = c("Bottom 20%", "Middle 60%", "Top 20%")) %>%
  mutate(Measure_Type = "HDB Resale Price")

# Combine
tidy_data <- bind_rows(income_long_final, hdb_long_final) %>%
  mutate(
    Income_Group = factor(Income_Group,
                          levels = c("Bottom 20%", "Middle 60%", "Top 20%")),
    Measure_Type = factor(Measure_Type,
                          levels = c("Real Income", "HDB Resale Price"))
  ) %>%
  select(Year, Income_Group, Measure_Type, Index_Value) %>%
  arrange(Year, Income_Group, Measure_Type)
